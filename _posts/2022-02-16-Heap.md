## deque, heapq 차이 및 사용법
> deque 
> - 선입선출
> - BFS 
> 
> from collections import deque    
> q = deque()     
> q.append('l')     
> q.popleft() 


> heapq
> - 최소힙, 최대힙
> - 최소값이나 최대값을 빨리 찾아야 할 때
>
> from heapq import heappush, heappop, heapify     
> q=[]     
> heappush(q,1)       
> heappop(q)    
> heapify(arr)
